      '',
      'Tasks:',
      '1. Optimize user description into a professional Agent description (Chinese)',
      '2. Generate a 4-10 char Chinese Agent name',
      '3. From the list above, select ONLY the tools truly needed (max 3)',
      '   - For web info (weather, news), use browser_navigate',
      '   - For sending email or executing scripts, use shell_run',
      '4. Generate a highly detailed and professional System Prompt (instruction) for the AI Agent. This prompt MUST instruct the Agent how to logically coordinate the selected tools step-by-step to achieve the user goal. (Chinese)',

      '{"name":"name","description":"desc","instruction":"System Prompt here...","tools":["tool1","tool2"]}',
    ].join('\n');

    let selectedTools: string[] = [];
    let parsedInstruction = '';
    
    try {
      const aiResult: string = await invoke('chat_with_ai', {
        req: { prompt: aiPrompt, system_prompt: 'AI Agent design expert. Return pure JSON only, no markdown. Only select from given tool list.', module: 'agent' }
      });
      const jsonMatch = aiResult.match(/\{[\s\S]*\}/);
      if (jsonMatch) {
        const parsed = JSON.parse(jsonMatch[0]);
        if (parsed.name) setAgentName(parsed.name);
        if (parsed.description) setAgentDesc(parsed.description);
        if (parsed.instruction) {
          parsedInstruction = parsed.instruction;
          setSystemInstruction(parsed.instruction);
        }
        if (Array.isArray(parsed.tools)) {
          const validNames = new Set(AVAILABLE_TOOLS.map(t => t.name));
          selectedTools = parsed.tools.filter((t: string) => validNames.has(t));
        }
      }
    } catch (aiErr) {
